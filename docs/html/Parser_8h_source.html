210 int countHeaders(const std::string& header) {
211 std::stringstream ss(header);
212 std::string word;
213 int count = 0;
214 while (std::getline(ss, word, ',')) {
215 for (char& c : word) if (isdigit(c)) return -1;
216 if (count == -1) break;
217 count++;
218 }
219
220 if (std::getline(ss, word, '\r')) {
221 for (char& c : word) if (isdigit(c)) return -1;
222
223 count++;
224 }
225 return count;

450 int parseTrips() {
451 std::ifstream file(trips_file);
452 if (!file.is_open()) {
453 std::cout << "Error opening file" << std::endl;
454 return -1;
455 }
456 auto stringToLower = [&](std::string& label) {
457 if (!label.empty()) { // Check if string is not empty
458 for (char& c : label) c = (char)tolower(c);
459 }
460 };
461 std::string line;
462 std::getline(file, line);
463 stringToLower(line);
464 if (countHeaders(line) < 0) return -1;
465
466 while (std::getline(file, line)) {
467 if (line.empty()) continue;
468
469 if (line.back() == '\r') {
470 line.pop_back();
471 }
472
473 std::istringstream s(line);
474 std::string route_id,direction_id,service_id, trip_id,trip_headsign, unused, shape_id, wheelchair;
475 // Read all fields with proper error checking
476 if (!std::getline(s, route_id, ',') || route_id.empty() ||
477 !std::getline(s, direction_id, ',') || direction_id.empty() ||
478 !std::getline(s, service_id, ',') || service_id.empty() ||
479 !std::getline(s, trip_id, ',') || trip_id.empty() ||
480 !std::getline(s, trip_headsign, ',') ||
481 !std::getline(s, unused, ',') || // wheelchair_accessible
482 !std::getline(s, unused, ',') || // block_id
483 !std::getline(s, shape_id)) { // shape_id
484 std::cerr << "Error parsing line: " << line << std::endl;
485 continue; // Skip malformed line but continue with others
486 }
487
488
489 stringToLower(route_id);
490 stringToLower(service_id);
491 stringToLower(trip_headsign);
492 stringToLower(shape_id);
493 stringToLower(trip_id);
494
495 try {
496 Trips trip(trip_id, route_id, std::stoi(direction_id), service_id, trip_headsign, shape_id);
497 insertTrip(trip_id, trip);
498 } catch (const std::exception& e) {
499 std::cerr << "Error creating trip from line: " << line << "\nError: " << e.what() << std::endl;
500 continue;
501 }
502
503 }
504 file.close();
505 return 0;

517 void parseVehicles(const std::string &jsonFile) {
518 std::ifstream file(jsonFile);
519 if (!file) {
520 throw std::runtime_error("Unable to open the JSON file: " + jsonFile);
521 }
522
523 // Parse JSON file using nlohmann::json
524 nlohmann::json jsonData;
525 file >> jsonData;
526
527 for (const auto &busEntry : jsonData) {
528 Vehicle vehicle;
529
530 // Parse annotations for route, trip, and direction
531 const auto annotations = busEntry["annotations"]["value"];
532 for (const auto &annotation : annotations) {
533 std::string annotationStr = annotation.get<std::string>();
534
535 if (annotationStr.find("stcp%3Aroute%3A") != std::string::npos) {
536 size_t pos = annotationStr.rfind("%3A");
537 if (pos != std::string::npos) {
538 std::string routeIdStr = annotationStr.substr(pos + 3);
539 try {
540 vehicle.setRouteId(std::stoi(routeIdStr));
541 } catch (const std::invalid_argument &e) {
542 //std::cerr << "Invalid route ID: " << routeIdStr << std::endl;
543 } catch (const std::out_of_range &e) {
544 std::cerr << "Route ID out of range: " << routeIdStr << std::endl;
545 }
546 }
547 } else if (annotationStr.find("stcp%3Anr_viagem%3A") != std::string::npos) {
548 size_t pos = annotationStr.find_last_of("%3A");
549 if (pos != std::string::npos) {
550 std::string tripStr = annotationStr.substr(pos + 1);
551 try {
552 vehicle.setTrip(std::stoi(tripStr));
553 } catch (const std::invalid_argument &e) {
554 //std::cerr << "Invalid trip number: " << tripStr << std::endl;
555 } catch (const std::out_of_range &e) {
556 std::cerr << "Trip number out of range: " << tripStr << std::endl;
557 }
558 }
559 } else if (annotationStr.find("stcp%3Asentido%3A") != std::string::npos) {
560 size_t pos = annotationStr.find_last_of("%3A");
561 if (pos != std::string::npos) {
562 std::string directionStr = annotationStr.substr(pos + 1);
563 try {
564 vehicle.setDirection(std::stoi(directionStr));
565 } catch (const std::invalid_argument &e) {
566 std::cerr << "Invalid direction: " << directionStr << std::endl;
567 } catch (const std::out_of_range &e) {
568 std::cerr << "Direction out of range: " << directionStr << std::endl;
569 }
570 }
571 }
572 }
573
574 // Parse agency (data provider)
575 if (busEntry.contains("dataProvider")) {
576 vehicle.setAgencyId(busEntry["dataProvider"]["value"]);
577 }
578
579 // Parse location (latitude and longitude)
580 if (busEntry.contains("location")) {
581 const auto location = busEntry["location"]["value"];
582 const auto coordinates = location["coordinates"];
583 Coordinates newcoordinates = {coordinates[1].get<double>(),coordinates[0].get<double>()};
584 vehicle.setCoordinates(newcoordinates);
585 }
586
587 // Add vehicle to the collection
588 vehicles.push_back(vehicle);
589 }
