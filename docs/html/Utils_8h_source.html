76 static double vincentyFormula(Coordinates coordinate1, Coordinates coordinate2) {
77 // Convert degrees to radians
78 double lat1 = coordinate1.Latitude;
79 double lat2 = coordinate2.Latitude;
80 double lon1 = coordinate1.Longitude;
81 double lon2 = coordinate2.Longitude;
82 double U1 = atan((1 - FLATTENING) * tan(lat1 * M_PI / 180.0));
83 double U2 = atan((1 - FLATTENING) * tan(lat2 * M_PI / 180.0));
84 double L = (lon2 - lon1) * M_PI / 180.0;
85 double lambda = L;
86
87 double sinU1 = sin(U1), cosU1 = cos(U1);
88 double sinU2 = sin(U2), cosU2 = cos(U2);
89
90 double sinLambda, cosLambda, sinSigma, cosSigma, sigma, sinAlpha, cos2Alpha, cos2SigmaM;
91 double lambdaPrev;
92
93 int iterations = 0;
94 do {
95 sinLambda = sin(lambda);
96 cosLambda = cos(lambda);
97 sinSigma = sqrt((cosU2 * sinLambda) * (cosU2 * sinLambda) +
98 (cosU1 * sinU2 - sinU1 * cosU2 * cosLambda) * (cosU1 * sinU2 - sinU1 * cosU2 * cosLambda));
99 if (sinSigma == 0) {
100 return 0; // Coincident points
101 }
102 cosSigma = sinU1 * sinU2 + cosU1 * cosU2 * cosLambda;
103 sigma = atan2(sinSigma, cosSigma);
104 sinAlpha = cosU1 * cosU2 * sinLambda / sinSigma;
105 cos2Alpha = 1 - sinAlpha * sinAlpha;
106 cos2SigmaM = cosSigma - 2 * sinU1 * sinU2 / cos2Alpha;
107 double C = FLATTENING / 16 * cos2Alpha * (4 + FLATTENING * (4 - 3 * cos2Alpha));
108 lambdaPrev = lambda;
109 lambda = L + (1 - C) * FLATTENING * sinAlpha *
110 (sigma + C * sinSigma * (cos2SigmaM + C * cosSigma * (-1 + 2 * cos2SigmaM * cos2SigmaM)));
111 } while (fabs(lambda - lambdaPrev) > 1e-12 && ++iterations < 100);
112
113 if (iterations >= 100) {
114 return -1; // Formula failed to converge
115 }
116
117 double uSquared = cos2Alpha * (MAJOR_AXIS * MAJOR_AXIS - MINOR_AXIS * MINOR_AXIS) / (MINOR_AXIS * MINOR_AXIS);
118 double A = 1 + uSquared / 16384 * (4096 + uSquared * (-768 + uSquared * (320 - 175 * uSquared)));
119 double B = uSquared / 1024 * (256 + uSquared * (-128 + uSquared * (74 - 47 * uSquared)));
120 double deltaSigma = B * sinSigma * (cos2SigmaM + B / 4 * (cosSigma * (-1 + 2 * cos2SigmaM * cos2SigmaM) -
121 B / 6 * cos2SigmaM * (-3 + 4 * sinSigma * sinSigma) * (-3 + 4 * cos2SigmaM * cos2SigmaM)));
122
123 double s = MINOR_AXIS * A * (sigma - deltaSigma);
124 return s; // Distance in meters
125 }

182 static MatrixXd generateMeasurements(unsigned int maxSamples, double totalDistance) {
183 MatrixXd measurements(maxSamples, 3);
184 measurements.setZero();
185
186 // Generate realistic bus movement profile
187 double step = totalDistance / (maxSamples-1);
188 double current_vel = 0;
189 double max_vel = 15.0; // ~54 km/h
190
191 for (unsigned int i = 0; i < maxSamples; i++) {
192 // Simulate bus acceleration then deceleration
193 if (i < maxSamples/2) {
194 current_vel = std::min(max_vel, current_vel + 0.1);
195 } else {
196 current_vel = std::max(0.0, current_vel - 0.1);
197 }
198
199 // Add realistic noise (5m max error)
200 double noise = (rand() % 1000)/100.0 - 5.0;
201
202 measurements(i, 0) = i * step + noise;
203 measurements(i, 1) = current_vel + noise/10.0;
204 measurements(i, 2) = 0; // Near-zero acceleration
205 }
206
207 return measurements;
208 }

221 static double calculateBusDistance(const std::map<std::string,std::vector<Coordinates>>& tripShape,
222 const std::string& shape_id,
223 const Coordinates& stop_coordinates) {
224 try {
225 double bus_distance = 0;
226 double distance = 0;
227 double shortest_distance = 500000;
228 auto shape_it = tripShape.find(shape_id);
229 if (shape_it == tripShape.end()) {
230 throw std::runtime_error("Shape ID not found");
231 }
232
233 const auto& coordinates = shape_it->second;
234 if (coordinates.empty()) {
235 throw std::runtime_error("No coordinates for shape");
236 }
237
238 for (size_t i = 1; i < coordinates.size(); i++) {
239 bus_distance += vincentyFormula(coordinates[i - 1], coordinates[i]);
240 distance = vincentyFormula(coordinates[i], stop_coordinates);
241 if (distance < shortest_distance) { shortest_distance = distance; }
242 //Calculates distance up until 200m of the bus stop
243 //std::cout << "Distance: " << distance << std::endl;
244 }
245
246 if (distance > 0) return distance;
247 std::ostringstream error_msg;
248 error_msg << "Stop not found within route " << shortest_distance;
249 throw std::runtime_error(error_msg.str());
250 } catch (const std::exception& e) {
251 std::cerr << "Error in calculateBusDistance: " << e.what() << std::endl;
252 return -1.0;
253 }
254 }

264 static std::map<std::string, double> calculateTripDistances(const std::map<std::string, std::vector<Coordinates>> &trips) {
265 std::map<std::string, double> tripDistances;
266
267 for (const auto &[shape_id, coordinates] : trips) {
268 double totalDistance = 0.0;
269 for (size_t i = 1; i < coordinates.size(); i++) {
270 totalDistance += vincentyFormula(coordinates[i - 1], coordinates[i]);
271 }
272 tripDistances[shape_id] = totalDistance;
273 }
274
275 return tripDistances;
